<script>

 const $ = id => document.getElementById(id);

 

 function fmt(n) {

   if (n >= 1000000) return (n/1000000).toFixed(1).replace(/\.0$/,'') + ' млн';

   if (n >= 1000) return Math.round(n/1000) + ' тыс';

   return Math.round(n).toString();

 }

 

 function calc() {

   const goal = parseFloat($('goal').value) || 0;

   const check = parseFloat($('check').value) || 1;

   const conv = parseFloat($('conv').value) / 100;

   const leadConv = parseFloat($('lead-conv').value) / 100;

   const repeat = parseFloat($('repeat').value) / 100;

   const cur = $('currency').value;

 

   const dealsNeeded = goal / check;

   const newDealsNeeded = dealsNeeded * (1 - repeat);

   const leadsNeeded = newDealsNeeded / conv;

   const trafficNeeded = leadsNeeded / leadConv;

   const dailyLeads = leadsNeeded / 22;

 

   $('r-deals').textContent = Math.ceil(dealsNeeded);

   $('r-leads').textContent = Math.ceil(leadsNeeded);

   $('r-traffic').textContent = fmt(Math.ceil(trafficNeeded));

   $('r-daily').textContent = Math.ceil(dailyLeads);

 

   $('fn-traffic').textContent = fmt(Math.ceil(trafficNeeded));

   $('fn-leads').textContent = Math.ceil(leadsNeeded);

   $('fn-deals').textContent = Math.ceil(dealsNeeded);

   $('fn-rev').textContent = fmt(goal) + ' ' + cur;

 

   const max = trafficNeeded;

   $('bar-traffic').style.width = '100%';

   $('bar-leads').style.width = Math.max(4, (leadsNeeded/max)*100) + '%';

   $('bar-deals').style.width = Math.max(2, (dealsNeeded/max)*100) + '%';

   $('bar-rev').style.width = Math.max(2, (dealsNeeded/max)*100) + '%';

 

   const leadsPerDeal = Math.round(leadsNeeded / Math.max(1, dealsNeeded));

   let insight = '';

   if (conv < 0.05) {

     insight = `Твоя конверсия в продажу <strong>${Math.round(conv*100)}%</strong> — это очень низко. Главный рычаг сейчас не трафик, а качество лидов и скрипт продаж. Подними конверсию до 15-20% — и нужный охват упадёт в разы.`;

   } else if (leadConv < 0.05) {

     insight = `Конверсия из охвата в лид <strong>${Math.round(leadConv*100)}%</strong> говорит о слабом оффере. Люди видят предложение, но не реагируют. Попробуй изменить формулировку боли или добавить конкретный результат.`;

   } else if (repeat > 0.3) {

     insight = `Отлично — у тебя <strong>${Math.round(repeat*100)}%</strong> повторных покупок. Это сильный актив. Фокус на удержании работает лучше чем постоянный поиск новых клиентов.`;

   } else {

     insight = `Чтобы выйти на <strong>${fmt(goal)} ${cur}</strong>, нужно касаться примерно <strong>${fmt(Math.ceil(trafficNeeded))} человек</strong> в месяц. На каждую сделку приходится <strong>${leadsPerDeal} лидов</strong>. Главный вопрос: откуда возьмёшь этот трафик и как будешь прогревать?`;

   }

   $('insight-text').innerHTML = insight;

 }

 

 $('conv').addEventListener('input', e => { $('conv-val').textContent = e.target.value; calc(); });

 $('lead-conv').addEventListener('input', e => { $('lead-conv-val').textContent = e.target.value; calc(); });

 $('repeat').addEventListener('input', e => { $('repeat-val').textContent = e.target.value; calc(); });

 $('goal').addEventListener('input', calc);

 $('check').addEventListener('input', calc);

 $('currency').addEventListener('change', calc);

 

 calc();

</script>
